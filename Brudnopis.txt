int wartoscMax(int a, int b, int c){
  int max = a;
    if (b>max) {
      max = b;
    }
    if (c>max) {
      max = c;
    }

  return max;
}



int tablicaMax(int tab[], int n) {
  int max = tab[0];
  for(int i=1; i<n; i++) {
   if (tab[i]>max) {
     max = tab[i];
   }
  }

  return max;
}

int tablicaSuma(int tab[], int n) {
  int suma = 0;
  for (int i = 0; i<n; i++) {
    suma += tab[i];
  }
  return suma;
}
